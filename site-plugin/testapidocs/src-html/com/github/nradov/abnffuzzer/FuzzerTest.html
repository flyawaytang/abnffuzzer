001package com.github.nradov.abnffuzzer;
002
003import static org.junit.Assert.assertArrayEquals;
004import static org.junit.Assert.assertEquals;
005import static org.junit.Assert.assertTrue;
006
007import java.io.ByteArrayOutputStream;
008import java.io.File;
009import java.io.FileOutputStream;
010import java.io.IOException;
011import java.io.OutputStream;
012import java.io.PrintStream;
013import java.io.PrintWriter;
014import java.nio.charset.StandardCharsets;
015import java.security.NoSuchAlgorithmException;
016import java.security.SecureRandom;
017import java.util.Arrays;
018import java.util.Collections;
019import java.util.HashSet;
020import java.util.Random;
021
022import org.apache.commons.cli.ParseException;
023import org.junit.Test;
024
025import com.github.nradov.abnffuzzer.Fuzzer;
026
027/**
028 * Test the {@link Fuzzer} class.
029 *
030 * @author Nick Radov
031 */
032public class FuzzerTest {
033
034    @Test
035    public void testMain() throws ParseException, IOException {
036        final File input = File.createTempFile(FuzzerTest.class.getSimpleName(),
037                null);
038        input.deleteOnExit();
039        try (final PrintWriter pw = new PrintWriter(
040                new FileOutputStream(input))) {
041            pw.println("foo = \"foo\" / bar\n");
042            pw.flush();
043        }
044        final File output = File
045                .createTempFile(FuzzerTest.class.getSimpleName(), null, null);
046        output.deleteOnExit();
047        Fuzzer.main(new String[] { "--count=10", "--separator=_",
048                "--input=" + input.getAbsolutePath(),
049                "--output=" + output.getAbsolutePath(),
050                "--charset=" + StandardCharsets.US_ASCII, "--exclude=bar",
051                "foo" });
052        assertEquals(10 * "foo".length() + 9, output.length());
053
054        OutputStream os = new ByteArrayOutputStream();
055        System.setOut(new PrintStream(os));
056        Fuzzer.main(new String[] { "-?" });
057        assertTrue(os.toString().length() > 0);
058
059        os = new ByteArrayOutputStream();
060        System.setOut(new PrintStream(os));
061        Fuzzer.main(new String[] { "-help" });
062        assertTrue(os.toString().length() > 0);
063    }
064
065    @Test
066    public void testGenerateAscii() throws IOException {
067        final Fuzzer f = new Fuzzer("foo = \"foo\"\n");
068        assertEquals("foo", f.generateAscii("foo"));
069    }
070
071    @Test(expected = IllegalArgumentException.class)
072    public void testGenerateAsciiExclude() throws IOException {
073        final Fuzzer f = new Fuzzer("foo = bar / baz\n");
074        f.generateAscii("foo", new HashSet<String>(
075                Arrays.asList(new String[] { "bar", "baz" })));
076    }
077
078    @Test
079    public void testGenerate() throws IOException {
080        final Fuzzer f = new Fuzzer("foo = \"foo\"\n");
081        assertArrayEquals(new byte[] { 'f', 'o', 'o' }, f.generate("foo"));
082    }
083
084    @Test
085    public void testGenerateCharset() throws IOException {
086        final Fuzzer f = new Fuzzer("foo = %x00.66.00.6F.00.6F");
087        assertEquals("foo", f.generate("foo", Collections.emptySet(),
088                StandardCharsets.UTF_16));
089    }
090
091    @Test
092    public void testSetRandom() throws NoSuchAlgorithmException, IOException {
093        final Random r = SecureRandom.getInstanceStrong();
094        final Fuzzer f = new Fuzzer("");
095        f.setRandom(r);
096        assertEquals(r, f.getRandom());
097    }
098
099}
