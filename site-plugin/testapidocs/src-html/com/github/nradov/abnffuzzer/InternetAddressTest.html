001package com.github.nradov.abnffuzzer;
002
003import java.io.IOException;
004import java.net.URISyntaxException;
005import java.util.Collections;
006import java.util.HashSet;
007import java.util.Set;
008
009import javax.mail.internet.AddressException;
010import javax.mail.internet.InternetAddress;
011
012import org.junit.Test;
013
014import com.github.nradov.abnffuzzer.Fuzzer;
015
016/**
017 * Test the {@link InternetAddress} class.
018 *
019 * @author Nick Radov
020 */
021public class InternetAddressTest {
022
023    @SuppressWarnings("serial")
024    private static final Set<String> EXCLUDE_2822 = Collections
025            .unmodifiableSet(new HashSet<String>() {
026                {
027                    add("CFWS");
028                    add("comment");
029                    add("quoted-string");
030                    add("obs-local-part");
031                }
032            });
033
034    @Test
035    public void testInternetAddress()
036            throws URISyntaxException, IOException, AddressException {
037        final Fuzzer rfc2822 = new Fuzzer(
038                InternetAddressTest.class.getResourceAsStream("/rfc2822.txt"));
039        final Fuzzer rfc1034 = new Fuzzer(
040                InternetAddressTest.class.getResourceAsStream("/rfc1034.txt"));
041
042        for (int i = 0; i < 1000; i++) {
043            final String localPart = rfc2822.generateAscii("local-part",
044                    EXCLUDE_2822);
045            final String domain = rfc1034.generateAscii("domain").trim();
046            if (domain.isEmpty()) {
047                // legal but not supported
048                continue;
049            }
050            final InternetAddress addr = new InternetAddress(
051                    localPart + "@" + domain, true);
052            addr.validate();
053        }
054    }
055
056}
