001package com.github.nradov.abnffuzzer;
002
003import java.io.IOException;
004import java.net.URI;
005import java.net.URISyntaxException;
006import java.util.Collections;
007import java.util.HashSet;
008import java.util.Set;
009
010import javax.mail.internet.AddressException;
011
012import org.junit.Test;
013
014import com.github.nradov.abnffuzzer.Fuzzer;
015
016/**
017 * Test the {@link URI} class.
018 *
019 * @author Nick Radov
020 */
021public class UriTest {
022
023    @SuppressWarnings("serial")
024    private static final Set<String> EXCLUDE_2396 = Collections
025            .unmodifiableSet(new HashSet<String>() {
026                {
027                    add("IPvFuture"); // not supported
028                    add("path-empty"); // not supported
029                    add("path-abempty"); // not supported
030                }
031            });
032
033    @Test
034    public void testUri()
035            throws URISyntaxException, IOException, AddressException {
036        final Fuzzer f = new Fuzzer(
037                UriTest.class.getResourceAsStream("/rfc3986.txt"));
038
039        final Set<String> tried = new HashSet<>();
040        for (int i = 0; i < 100; i++) {
041            final String uri = f.generateAscii("URI-reference", EXCLUDE_2396);
042            if (!tried.contains(uri)) {
043                try {
044                    new URI(uri);
045                } catch (final URISyntaxException e) {
046                    // work around differences between RFC 3986 and RFC 2396
047                    if (!(e.getMessage()
048                            .startsWith("Expected authority at index")
049                            || e.getMessage().startsWith(
050                                    "Malformed port number at index"))) {
051                        System.err.println(e.getMessage());
052                        System.err.println(uri);
053                        System.err.println();
054                        throw e;
055                    }
056                }
057                tried.add(uri);
058            }
059        }
060    }
061
062}
